import matplotlib.pyplot as plt 
import seaborn as sns
print('Column: Velocidad horario del viento (m/s)')
print('Skew:', round(wind_speed['Velocidad horario del viento (m/s)'].skew(), 2))
print('Kurtosis: ', round(wind_speed['Velocidad horario del viento (m/s)'].kurtosis(), 2))
plt.figure(figsize = (14, 6))
plt.subplot(1, 2, 1)
#wind_speed['Velocidad horario del viento (m/s)'].hist(grid=False)
sns.distplot(wind_speed['Velocidad horario del viento (m/s)'], axlabel="Velocidad horario del viento");
plt.title('Historgrama de Velocidad horario del viento')
plt.subplot(1, 2, 2)
sns.boxplot(wind_speed['Velocidad horario del viento (m/s)'])
plt.title('Boxplot de Velocidad horario del viento')
plt.show()

Column: Velocidad horario del viento (m/s)
Skew: 0.37
Kurtosis:  -0.11

Basado en los resultados podemos identificar que la variable ‘velocidad horario del viento’:

Asimetria: (Skew: 0.37) Presenta un ligero sesgo positivo, esto quiere decir que tiene mayor peso en la cola izquierda de la distribucion, tal y como se obserba en el histograma.

Kurtosis: (Kurtosis: -0.11) La forma de la distribucion de frecuencias es platicurtica, esto significa que la distribucion presenta colas menos gruesas que la normal.

from sklearn.model_selection import TimeSeriesSplit, GridSearchCV
from sklearn.metrics import make_scorer, r2_score, mean_squared_error
import numpy as np

# RMSE (Root Mean Squared Error)
def rmse(y_true, y_pred):
    return np.sqrt(np.mean((y_true - y_pred) ** 2))

# MAPE (Mean Absolute Percentage Error)
def mape(y_true, y_pred):     
    mask = y_true != 0
    return np.mean(np.abs((y_true - y_pred) / y_true))*100


X = wind_speed_X.tail(24*6)
y = wind_speed_y.tail(24*6)


def gridSearchRegresor(model, param):
    tscv = TimeSeriesSplit(n_splits=5,test_size=24,gap=0)
    gsearch = GridSearchCV(estimator=model, 
                            cv=tscv, 
                            param_grid=param, 
                            scoring = { 'R2': make_scorer(r2_score),
                                        'RMSE':make_scorer(rmse,greater_is_better=False), 
                                        'MAPE':make_scorer(mape,greater_is_better=False)}, 
                            refit = 'RMSE',
                            #verbose=2
                            )
    gsearch.fit(X, y)
    results = gsearch.cv_results_
    results
    # Extract results for multiple scores
    R2_results = results['mean_test_R2']
    RMSE_results = results['mean_test_RMSE']*-1
    MAPE_results = results['mean_test_MAPE']*-1
    
    dic={"R2":R2_results[RMSE_results.argmin()],"MAPE":MAPE_results[RMSE_results.argmin()],"RMSE":RMSE_results[RMSE_results.argmin()]}

    return dic
